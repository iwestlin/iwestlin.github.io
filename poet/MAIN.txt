procedure TFormMain.ButtonMakeClick(Sender: TObject);
var
  i,j: Integer;
  s:string;
  r: Integer;
  b: Boolean;
begin
  // 生成诗结构
  cpjgd.Clear;
  MemoMain.Clear;
  Randomize;
  if ComboBoxFD.Text='是'
  then
  begin
    hsb:=strtoint(LabeledEditHSB.Text);
    dsb:=strtoint(LabeledEditDSB.Text);
    for i:=0 to hsb-1 do
      cpjgd.Append(inttostr(Random(137)));
    cpjgd.Append('');
    for i:=1 to dsb-1 do
      for j:=0 to hsb do
        cpjgd.Append(cpjgd.Strings[j]);
  end
  else
  begin
    hs:=strtoint(LabeledEditHS.Text);
    Randomize;
    for i:=0 to hs-1 do
      cpjgd.Append(inttostr(Random(137)));
  end;
  // 生成诗
  for i:=0 to cpjgd.Count-1 do
  begin
    if length(cpjgd.Strings[i])>0
    then
      s:=cpjg[strtoint(cpjgd.Strings[i])]
    else
      s:='';
    for j:=length(s)-1 downto 0 do
    begin
      if copy(s,j,2)='MM'
      then
      begin
        b:=True;
        while b do
        begin
          r:=Random(1598);
          if ((ComboBoxYL.Text='是') and (ComboBoxYJB.Text=mmc[r])) or (ComboBoxYL.Text='否')
          then
            b:=False;
        end;
        s:=copy(s,0,j-1)+mma[r]+copy(s,j+2,length(s));
      end;
      if copy(s,j,2)='DD'
      then
      begin
        b:=True;
        while b do
        begin
          r:=Random(324);
          if ((ComboBoxYL.Text='是') and (ComboBoxYJB.Text=ddc[r])) or (ComboBoxYL.Text='否')
          then
            b:=False;
        end;
        s:=copy(s,0,j-1)+dda[r]+copy(s,j+2,length(s));
      end;
      if copy(s,j,2)='XX'
      then
      begin
        b:=True;
        while b do
        begin
          r:=Random(517);
          if ((ComboBoxYL.Text='是') and (ComboBoxYJB.Text=xxc[r])) or (ComboBoxYL.Text='否')
          then
            b:=False;
        end;
        s:=copy(s,0,j-1)+xxa[r]+copy(s,j+2,length(s));
      end;
      if copy(s,j,2)='DJ'
      then
      begin
        b:=True;
        while b do
        begin
          r:=Random(230);
          if ((ComboBoxYL.Text='是') and (ComboBoxYJB.Text=djc[r])) or (ComboBoxYL.Text='否')
          then
            b:=False;
        end;
        s:=copy(s,0,j-1)+dja[r]+copy(s,j+2,length(s));
      end;
      if copy(s,j,2)='TT'
      then
        s:=copy(s,0,j-1)+tt[Random(15)]+copy(s,j+2,length(s));
    end;
    MemoMain.Lines.Append(s);
  end;
end;

procedure TFormMain.ButtonSaveClick(Sender: TObject);
var
  Title: string;
  f:textfile;
  i: Integer;
begin
  Title:=InputBox('保存','诗的名字：','');
  if length(Title)>0
  then
  begin
    AssignFile(f,Title+'.txt') ;
    Rewrite(f);
    writeln(f,Title);
    writeln(f,'');
    writeln(f,'作者：计算机诗人　程序设计：刘慈欣　移植：sft');
    writeln(f,'');
    for i:=0 to MemoMain.Lines.Count-1 do
      writeln(f,MemoMain.Lines[i]);
    writeln(f,'');
    writeln(f,'中国科幻：http://www.chinasf.com　科幻望远镜：http://sft@chinasf.com　Email：allenjiang@163.net');
    CloseFile(f);
    showmessage('诗已经保存在当前目录下！');
  end;
end;
